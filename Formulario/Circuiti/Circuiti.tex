\begin{gather*}
    \textbf{Differenza di Potenziale: } \\ V = I \cdot R \quad I = \frac{\Delta V}{R} \\
    \textbf{Potenza Dissipata: } \\
    W = I \cdot V = I^2 R = \frac{V^2}{R} \\
