\section{Vettori}

\begin{gather*}
    \vec{v} = \begin{cases}
        v_x = v \cdot \cos \alpha \\
        v_y = v \cdot \sin \alpha
    \end{cases} \\
    \vec{v} = \vec{v}_x + \vec{v}_y \\
    v = \sqrt{v_x^2 + v_y^2} \\
    \tan \theta = \frac{v_y}{v_x} \\
    \theta = \arctan (\frac{v_y}{v_x})
\end{gather*}
