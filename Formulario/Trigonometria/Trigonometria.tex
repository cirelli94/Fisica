\begin{gather*}
    \begin{cases}
        \sin \alpha = \frac{h}{l} \\
        \cos \alpha = \frac{d}{l} \\
        \tan \alpha = \frac{h}{d} \\
    \end{cases} \rightarrow
    \begin{cases}
        h = l \cdot \sin \alpha \\
        l = \frac{h}{\sin \alpha}
    \end{cases} \\
\end{gather*}
